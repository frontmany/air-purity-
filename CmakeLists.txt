cmake_minimum_required(VERSION 3.28.3 FATAL_ERROR)

project(air-purity)

set(QT_DRIVE "C")
SET(CMAKE_PREFIX_PATH  "${QT_DRIVE}:/Qt/6.5.1")
find_package(Qt6Widgets REQUIRED)


set(HEADERS include/mainwindow.h include/centralwidget.h include/menu.h)
include_directories(include)

set(SOURCES src/main.cpp src/mainwindow.cpp src/centralwidget.cpp src/menu.cpp)


add_executable(air-purity ${SOURCES} ${HEADERS})
target_link_libraries(air-purity Qt6::Widgets)
